```swift
import SwiftData

@Model
final class Food {
    var name: String = ""
    var categoryRawValue: String = "legumes"
    var recommendedAgeMonths: Int = 4
    var notes: String?

    var category: FoodCategory {
        get { FoodCategory(rawValue: categoryRawValue) ?? .legumes }
        set { categoryRawValue = newValue.rawValue }
    }

    @Relationship(deleteRule: .cascade, inverse: \FoodEntry.food)
    var entries: [FoodEntry]? = []

    init(name: String, category: FoodCategory, recommendedAgeMonths: Int, notes: String? = nil) {
        self.name = name
        self.categoryRawValue = category.rawValue
        self.recommendedAgeMonths = recommendedAgeMonths
        self.notes = notes
    }
}
```

```swift
import Foundation

@Model
final class FoodEntry {
    var dateIntroduced: Date = Date.now
    var reactionRawValue: String = "neutral"
    var hasDigestiveIssue: Bool = false
    var hasAllergicReaction: Bool = false
    var notes: String?
    var food: Food?

        get { Reaction(rawValue: reactionRawValue) ?? .neutral }

    init(food: Food, dateIntroduced: Date = .now, reaction: Reaction = .neutral, hasDigestiveIssue: Bool = false, hasAllergicReaction: Bool = false, notes: String? = nil) {

        self.reactionRawValue = reaction.rawValue
```

```swift
import SwiftData
```

```swift
@Observable
@MainActor
final class FoodsViewModel {
    var foods: [Food] = []
    var testedFoodIDs: Set<PersistentIdentifier> = []
    var selectedCategory: FoodCategory?
    var selectedAgeFilter: Int?
    var showOnlyTested: Bool?
    var showAddSheet = false
    var foodToDelete: Food?

    private let context: ModelContext

    init(context: ModelContext) {
        self.context = context
    }

    var filteredFoods: [Food] {
        foods.filter { food in
            if let cat = selectedCategory, food.category != cat { return false }
            if let age = selectedAgeFilter, food.recommendedAgeMonths > age { return false }
            if let tested = showOnlyTested {
                let isTested = testedFoodIDs.contains(food.persistentModelID)
                if tested && !isTested { return false }
                if !tested && isTested { return false }
            }
            return true
        }
    }

    var groupedFoods: [(FoodCategory, [Food])] {
        let grouped = Dictionary(grouping: filteredFoods) { $0.category }
        return FoodCategory.allCases
            .compactMap { cat in
                guard let foods = grouped[cat], !foods.isEmpty else { return nil }
                return (cat, foods.sorted { $0.name < $1.name })
            }
    }

    func fetchFoods() {
        let descriptor = FetchDescriptor<Food>(sortBy: [SortDescriptor(\.categoryRawValue), SortDescriptor(\.name)])
        foods = (try? context.fetch(descriptor)) ?? []

        let entryDescriptor = FetchDescriptor<FoodEntry>()
        let entries = (try? context.fetch(entryDescriptor)) ?? []
        testedFoodIDs = Set(entries.compactMap { $0.food?.persistentModelID })
    }

    func addFood(name: String, category: FoodCategory, ageMonths: Int, notes: String?) {
        let food = Food(name: name, category: category, recommendedAgeMonths: ageMonths, notes: notes)
        context.insert(food)
        try? context.save()
        fetchFoods()
    }

    func deleteFood(_ food: Food) {
        context.delete(food)
        try? context.save()
        fetchFoods()
    }

    func toggleCategory(_ category: FoodCategory) {
        selectedCategory = selectedCategory == category ? nil : category
    }

    func toggleAgeFilter(_ age: Int) {
        selectedAgeFilter = selectedAgeFilter == age ? nil : age
    }

    func toggleTestedFilter(_ tested: Bool?) {
        showOnlyTested = showOnlyTested == tested ? nil : tested
    }
}
```

```swift
    var testedFoods = 0
    var categoryProgress: [CategoryProgress] = []
    var likedCount = 0
    var neutralCount = 0
    var dislikedCount = 0
    var alerts: [AlertItem] = []

    var progressPercent: Double {
        totalFoods > 0 ? Double(testedFoods) / Double(totalFoods) : 0
    }

    private let context: ModelContext

    init(context: ModelContext) {
        self.context = context
    }

    func fetchProgress() {
        let foodDescriptor = FetchDescriptor<Food>()
        let allFoods = (try? context.fetch(foodDescriptor)) ?? []
        totalFoods = allFoods.count

        let entryDescriptor = FetchDescriptor<FoodEntry>()
        let allEntries = (try? context.fetch(entryDescriptor)) ?? []
        let testedIDs = Set(allEntries.compactMap { $0.food?.persistentModelID })
        testedFoods = testedIDs.count

        categoryProgress = FoodCategory.allCases.map { cat in
            let total = allFoods.filter { $0.category == cat }.count
            let tested = allFoods.filter { $0.category == cat && testedIDs.contains($0.persistentModelID) }.count
            return CategoryProgress(category: cat, total: total, tested: tested)
        }

        likedCount = allEntries.filter { $0.reaction == .liked }.count
        neutralCount = allEntries.filter { $0.reaction == .neutral }.count
        dislikedCount = allEntries.filter { $0.reaction == .disliked }.count

        alerts = allEntries
            .filter { $0.hasDigestiveIssue || $0.hasAllergicReaction }
            .compactMap { entry in
                guard let food = entry.food else { return nil }
                return AlertItem(
                    id: entry.persistentModelID,
```

```swift
import SwiftData
import Foundation

@Observable
@MainActor
final class FoodDetailViewModel {
    var food: Food?
    var existingEntry: FoodEntry?
    var dateIntroduced: Date = .now
    var reaction: Reaction = .neutral
    var hasDigestiveIssue = false
    var hasAllergicReaction = false
    var notes = ""
    var isSaved = false

    var isAlreadyTested: Bool { existingEntry != nil }

    private let context: ModelContext

    init(context: ModelContext) {
        self.context = context
    }

    func loadFood(id: PersistentIdentifier) {
        food = context.model(for: id) as? Food
        guard let food else { return }

        let entries = food.entries ?? []
        if let entry = entries.first {
            existingEntry = entry
            dateIntroduced = entry.dateIntroduced
            reaction = entry.reaction
            hasDigestiveIssue = entry.hasDigestiveIssue
            hasAllergicReaction = entry.hasAllergicReaction
            notes = entry.notes ?? ""
        }
    }

    func save() {
        guard let food else { return }

        if let entry = existingEntry {
            entry.dateIntroduced = dateIntroduced
            entry.reaction = reaction
            entry.hasDigestiveIssue = hasDigestiveIssue
            entry.hasAllergicReaction = hasAllergicReaction
            entry.notes = notes.isEmpty ? nil : notes
        } else {
            let entry = FoodEntry(
                food: food,
                dateIntroduced: dateIntroduced,
                reaction: reaction,
                hasDigestiveIssue: hasDigestiveIssue,
                hasAllergicReaction: hasAllergicReaction,
                notes: notes.isEmpty ? nil : notes
            )
            context.insert(entry)
        }
        try? context.save()
        isSaved = true
    }

    func deleteEntry() {
        guard let entry = existingEntry else { return }
        context.delete(entry)
        try? context.save()
        isSaved = true
    }
}
```

```swift
import SwiftData
import Foundation

struct ScreenshotDataService: Sendable {
    static var isScreenshotMode: Bool {
        ProcessInfo.processInfo.arguments.contains("-screenshot")
    }

    @MainActor
    static func populate(context: ModelContext) {
        SeedDataService.seedIfNeeded(context: context)

        let descriptor = FetchDescriptor<Food>()
        guard let allFoods = try? context.fetch(descriptor) else { return }

        // Mark ~40% as tested with varied reactions
        let foodsToTest = allFoods.prefix(90)
        let reactions: [Reaction] = [.liked, .liked, .liked, .neutral, .neutral, .disliked]
        let calendar = Calendar.current

        for (index, food) in foodsToTest.enumerated() {
            let daysAgo = Int.random(in: 1...60)
            let date = calendar.date(byAdding: .day, value: -daysAgo, to: .now) ?? .now
            let reaction = reactions[index % reactions.count]

            let entry = FoodEntry(
                food: food,
                dateIntroduced: date,
                reaction: reaction,
                hasDigestiveIssue: index % 15 == 0,
                hasAllergicReaction: index % 25 == 0
            )
            context.insert(entry)
        }
        try? context.save()
    }
}
```

## Task 17: GitHub Actions CI

**Files:**
- Create: `.github/workflows/ci.yml`

**Step 1: Create CI workflow**

Write to `.github/workflows/ci.yml`:

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

jobs:
  build-and-test:
    name: Build & Test
    runs-on: [self-hosted, macOS, ARM64]
    timeout-minutes: 30

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Select iOS Simulator
        id: sim
        run: |
          SIMULATORS=("iPhone 17 Pro" "iPhone 17" "iPhone 16e" "iPhone 16 Pro" "iPhone 15 Pro" "iPhone 15")
          for device in "${SIMULATORS[@]}"; do
            if xcrun simctl list devices available | grep -q "$device"; then
              echo "device=$device" >> "$GITHUB_OUTPUT"
              echo "Selected simulator: $device"
              exit 0
            fi
          done
          echo "No suitable simulator found" && exit 1

      - name: Generate Xcode project
        run: xcodegen generate

      - name: Build
        run: |
          set -o pipefail
          xcodebuild build-for-testing \
            -project PetitesBouchees.xcodeproj \
            -scheme PetitesBouchees \
            -destination 'platform=iOS Simulator,name=${{ steps.sim.outputs.device }}' \
            -skipPackagePluginValidation \
            CODE_SIGN_IDENTITY="" \
            CODE_SIGNING_REQUIRED=NO \
            CODE_SIGNING_ALLOWED=NO \
            | xcpretty

      - name: Run tests
        run: |
          set -o pipefail
          xcodebuild test-without-building \
            -project PetitesBouchees.xcodeproj \
            -scheme PetitesBouchees \
            -destination 'platform=iOS Simulator,name=${{ steps.sim.outputs.device }}' \
            -only-testing:PetitesBoucheesTests \
            | xcpretty

      - name: Upload test results
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: test-results
          path: ~/Library/Developer/Xcode/DerivedData/**/Logs/Test/*.xcresult
          retention-days: 5
```
